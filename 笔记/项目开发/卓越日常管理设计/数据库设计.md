SET NAMES utf8mb4;
SET FOREIGN_KEY_CHECKS = 0;

-- ----------------------------
-- Table structure for activities
-- ----------------------------
DROP TABLE IF EXISTS `activities`;
CREATE TABLE `activities`  (
  `activity_id` bigint NOT NULL AUTO_INCREMENT COMMENT '活动ID',
  `activity_title` varchar(50) CHARACTER SET utf8mb3 COLLATE utf8mb3_general_ci NOT NULL COMMENT '活动标题',
  `activity_content` varchar(10000) CHARACTER SET utf8mb3 COLLATE utf8mb3_general_ci NULL DEFAULT NULL COMMENT '活动内容',
  `activity_type_id` bigint NOT NULL COMMENT '活动类型ID',
  `activity_start_time` datetime NOT NULL COMMENT '活动开始时间',
  `activity_end_time` datetime NOT NULL COMMENT '活动结束时间',
  `activity_progress` char(1) DEFAULT '1' COMMENT '活动进行状态(0活动已结束,1活动未开始,2活动进行中)',
  `activity_initiator` varchar(255) CHARACTER SET utf8mb3 COLLATE utf8mb3_general_ci NOT NULL COMMENT '活动发起人',
  `activity_initiator_id` varchar(20) CHARACTER SET utf8mb3 COLLATE utf8mb3_general_ci NOT NULL COMMENT '活动发起人ID',
  `activity_initaition_time` datetime NOT NULL COMMENT '活动发起时间',
  `activity_update_time` datetime NULL DEFAULT NULL COMMENT '活动更新时间',
  `activity_reviewer` varchar(255) CHARACTER SET utf8mb3 COLLATE utf8mb3_general_ci NULL DEFAULT NULL COMMENT '活动审核人',
  `activity_review_time` datetime NULL DEFAULT NULL COMMENT '活动审核时间',
  `activity_review_comment` varchar(255) CHARACTER SET utf8mb3 COLLATE utf8mb3_general_ci NULL DEFAULT NULL COMMENT '活动审核意见',
  `activity_review_status` char(1) CHARACTER SET utf8mb3 COLLATE utf8mb3_general_ci NOT NULL DEFAULT '2' COMMENT '活动审核状态(0未通过,1通过,2暂未审核)',
  PRIMARY KEY (`activity_id`) USING BTREE
) ENGINE = InnoDB AUTO_INCREMENT = 5 CHARACTER SET = utf8mb3 COLLATE = utf8mb3_general_ci COMMENT = '活动表' ROW_FORMAT = DYNAMIC;

SET FOREIGN_KEY_CHECKS = 1;

活动开始时间<系统时间< 活动结束时间 ，状态：活动进行中 2
活动结束时间 < 系统时间，状态：活动已结束0
